% Redifines chapter formatting commands
\makeatletter
\def\@makechapterhead#1{%
  { \parindent \z@ \normalfont
    \ifnum \c@secnumdepth >\m@ne
        \thechapter \space - \space        
    \fi
    \interlinepenalty\@M
    \centering \MakeUppercase{#1}\par\nobreak
    \vskip 12\p@
  }
}
\def\@makeschapterhead#1{%
  {\parindent \z@
    \normalfont
    \interlinepenalty\@M
    \centering \MakeUppercase{#1}\par\nobreak
    \vskip 12\p@
  }
}
\def\@makesection#1{%
  {\parindent \z@
    \normalfont
    \interlinepenalty\@M
    \centering \MakeUppercase{#1}\par\nobreak
    \vskip 12\p@
  }
}
